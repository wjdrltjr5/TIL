## Item 클래스
```java
public class Item{
    private String name;

    public Item(String name){
        this.name = name;
    }

    public String name(){
        return name;
    }
}

// Item을 보관할 수 있는 Room 클래스
public class Room{
    private Position position;
    private String name;
    private String description;
    private List<Item> items = new ArrayList<>();

    public Room(Position position, String name, String description, Item ... items){
        this.position = position;
        this.name = name;
        this.description = description;
        this.items.addAll(List.of(items));
    }

    public List<Item> items(){
        return Collections.unmodifiableList(items);
    }
    // room에서 item 제거
    public void remove(Item item){
        this.item.remove(item);
    }

    // room에 아이템 제거
    public void add(Item item){
        this.items.add(item);
    }
}

// Item을 보관할 수 있는 Player 클래스
public class Player{
    private WorldMap worldMap;
    private Position position;
    private List<Item> items = new ArrayList<>();

    public Player(WorldMap worldMap, Position position, Item ... items){
        this.worldMap = worldMap;
        this.position = position;
        this.items.addAll(List.of(items));
    }

    public List<Item> items(){
        return Collections.unmodifiableList(items);
    }

    public Optional(Item) find(String itemName){
        return items.stream()
                    .filter(item -> item.name().equalsIgnoreCase(itemName))
                    .findFirst();
    }
    
    // player에 item 추가
    public void add(Item item){
        this.item.add(item);
    }

    // player에서 item 제거
    public void remove(Item item){
        this.items.remove(item);
    }

}

// Game 클래스에 takeItem 메서드 구현
public class Game{
    private Player player;
    private CommandParser commandParser;
    private InputOutput io;
    private boolean running;

    private void takeItem(String itemName){
        player.currentRoom().find(itemName).ifPresentOrElse(
            item -> {
                player.currentRoom().remove(item);
                player.add(item);
                showTaken(itemName);
            },
            () -> showTakeFailed(itemName));
    }

    private void showTaken(String itemName){
        io.showLine(itemName + "을(를) 얻었습니다.");
    }

    private void showTakeFailed(String itemName){
        io.showLine(itemName + "을(를) 얻을 수 없습니다.");
    }

    private void dropItem(String itemName){
        player.find(itemName).ifPresentOrElse(
            item -> {
                player.remove(item);
                player.currentRoom().add(item)
                showDropped(itemName);
            },
            () -> showDropFailed(itemName));
    }

    private void showDropped(String itemName){
        io.showLine(itemName + "을(를) 버렸습니다.");
    }

    private void showDropFailed(String itemName){
        io.showLine(itemName, "을(를) 버릴 수 없습니다.");
    }

    public void showInventory(){
        io.showLine(
            player.item().stream()
                .map(Item::name)
                .collect(Collectors.joining(", ", "인벤토리 목록 : [ ", " ]")));
    }
}
```
